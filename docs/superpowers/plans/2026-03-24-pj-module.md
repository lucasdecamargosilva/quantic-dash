# PJ Module Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add four PJ (business finance) pages to the Quantic Dashboard — Receitas, Clientes, DRE, and Fluxo de Caixa — backed by Supabase, following the existing vanilla HTML/CSS/JS patterns.

**Architecture:** Four standalone HTML files sharing the existing sidebar/header pattern from `custos.html`. Each page loads `supabase-config.js` and `auth.js` via CDN, queries Supabase directly from inline `<script>`, and renders dynamic content via `innerHTML` with all user-supplied strings HTML-escaped. No build step.

**Tech Stack:** Vanilla HTML/CSS/JS, Supabase JS SDK v2 (CDN), Phosphor Icons `ph-fill` (CDN), Google Fonts Outfit.

**Spec:** `docs/superpowers/specs/2026-03-24-pj-module-design.md`

---

## Security Pattern (apply to all pages)

All user-supplied strings inserted into DOM via innerHTML must be escaped with:

```js
const esc = s => String(s ?? '').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
```

Use `esc()` on every field from Supabase before interpolating into template literals.

## Task 1: Create Supabase Tables

**Files:** No code files — SQL run in Supabase SQL Editor (project: jytsrxrmgvliyyuktxsd)

- [ ] **Step 1: Run table creation SQL**

```sql
CREATE TABLE IF NOT EXISTS pj_clientes (
  id         uuid primary key default gen_random_uuid(),
  nome       text not null,
  documento  text,
  email      text,
  telefone   text,
  ativo      boolean default true,
  created_at timestamptz default now()
);

CREATE TABLE IF NOT EXISTS pj_receitas (
  id               uuid primary key default gen_random_uuid(),
  cliente_id       uuid references pj_clientes(id),
  descricao        text,
  valor            numeric not null,
  valor_recebido   numeric default 0,
  data_emissao     date,
  data_vencimento  date not null,
  data_recebimento date,
  status           text check (status in ('open','received','cancelled')) default 'open',
  -- NOTE: 'overdue' is intentionally excluded — it is a JS-only display state computed at render time.
  --       Spec DDL includes 'overdue' for documentation; this CHECK reflects actual DB behavior.
  created_at       timestamptz default now()
);

CREATE TABLE IF NOT EXISTS pj_custos_lancados (
  id             uuid primary key default gen_random_uuid(),
  descricao      text not null,
  valor          numeric not null,
  data_pagamento date not null,
  categoria      text,
  pago           boolean default false,
  created_at     timestamptz default now()
);

CREATE TABLE IF NOT EXISTS pj_config (
  id      uuid primary key default gen_random_uuid(),
  user_id uuid references auth.users(id) not null,
  chave   text not null,
  valor   text not null,
  unique(user_id, chave)
);
```


```html
<script>
const esc = s => String(s ?? '').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
const fmt = v => 'R$\u00a0' + Number(v || 0).toLocaleString('pt-BR', { minimumFractionDigits: 2 });
const today = new Date(); today.setHours(0,0,0,0);

async function getSB() {
    await window.CONFIG_LOADED;
    return window.supabaseClient || window.supabase.createClient(window.SUPABASE_CONFIG.URL, window.SUPABASE_CONFIG.KEY);
}

async function loadClientes() {
    const sb = await getSB();
    const term = document.getElementById('searchInput').value.toLowerCase();
    const { data: clientes, error } = await sb.from('pj_clientes').select('*').eq('ativo', true).order('nome');
    if (error) { document.getElementById('clientesGrid').innerHTML = `<div class="empty-state"><p>Erro: ${esc(error.message)}</p></div>`; return; }
    const { data: receitas } = await sb.from('pj_receitas').select('cliente_id, valor, status, data_vencimento');
    const filtered = (clientes || []).filter(c => !term || esc(c.nome).toLowerCase().includes(term) || (c.documento || '').toLowerCase().includes(term));
    if (!filtered.length) {
        document.getElementById('clientesGrid').innerHTML = `<div class="empty-state"><i class="ph-fill ph-buildings"></i><p>Nenhum cliente encontrado.</p></div>`;
        return;
    }
    document.getElementById('clientesGrid').innerHTML = filtered.map(c => {
        const cRec = (receitas || []).filter(r => r.cliente_id === c.id);
        const totalFaturado = cRec.filter(r => r.status !== 'cancelled').reduce((s,r) => s + Number(r.valor), 0);
        const emAberto = cRec.filter(r => r.status === 'open').reduce((s,r) => s + Number(r.valor), 0);
        const vencido = cRec.filter(r => r.status === 'open' && new Date(r.data_vencimento) < today).reduce((s,r) => s + Number(r.valor), 0);
        const initials = esc(c.nome).split(' ').slice(0,2).map(w => w[0]).join('').toUpperCase();
        return `<div class="client-card">
            <div class="client-card-top">
                <div class="client-avatar">${initials}</div>
                <button class="btn-icon-sm" onclick="desativarCliente('${esc(c.id)}')"><i class="ph-fill ph-trash"></i></button>
            </div>
            <div class="client-name">${esc(c.nome)}</div>
            ${c.documento ? `<div class="client-doc">${esc(c.documento)}</div>` : ''}
            ${c.email ? `<div class="client-contact"><i class="ph-fill ph-envelope"></i> ${esc(c.email)}</div>` : ''}
            ${c.telefone ? `<div class="client-contact"><i class="ph-fill ph-phone"></i> ${esc(c.telefone)}</div>` : ''}
            <div class="client-stats">
                <div class="client-stat"><span class="client-stat-label">Total Faturado</span><span class="client-stat-value">${fmt(totalFaturado)}</span></div>
                <div class="client-stat"><span class="client-stat-label">Em Aberto</span><span class="client-stat-value">${fmt(emAberto)}</span>${vencido > 0 ? `<span class="client-stat-sub">${fmt(vencido)} vencido</span>` : ''}</div>
            </div>
        </div>`;
    }).join('');
}

async function desativarCliente(id) {
    if (!confirm('Desativar este cliente?')) return;
    const sb = await getSB();
    await sb.from('pj_clientes').update({ ativo: false }).eq('id', id);
    loadClientes();
}

document.getElementById('btnNovoCliente').addEventListener('click', () => {
    ['inputNome','inputDocumento','inputEmail','inputTelefone'].forEach(id => document.getElementById(id).value = '');
    document.getElementById('modalCliente').classList.add('active');
});
document.getElementById('btnCancelarCliente').addEventListener('click', () => document.getElementById('modalCliente').classList.remove('active'));
document.getElementById('searchInput').addEventListener('input', loadClientes);
document.getElementById('btnSalvarCliente').addEventListener('click', async () => {
    const nome = document.getElementById('inputNome').value.trim();
    if (!nome) { alert('Nome obrigatório.'); return; }
    const sb = await getSB();
    const { error } = await sb.from('pj_clientes').insert({ nome, documento: document.getElementById('inputDocumento').value.trim() || null, email: document.getElementById('inputEmail').value.trim() || null, telefone: document.getElementById('inputTelefone').value.trim() || null });
    if (error) { alert('Erro: ' + error.message); return; }
    document.getElementById('modalCliente').classList.remove('active');
    loadClientes();
});
document.getElementById('toggleSidebar')?.addEventListener('click', () => document.getElementById('sidebar').classList.toggle('collapsed'));
loadClientes();
</script>
```
